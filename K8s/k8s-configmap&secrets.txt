----------From-literal--------------------------------------------------------------------------------------
-----------create cm---------------------------

kubectl create cm dev --from-literal=userid=mahesh --from-literal=password=123

----------injecting to Pod---------------------
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: dep-1
spec:
  replicas: 3
  selector:
    matchLabels:
      env: dev
  template:
    metadata:
      labels:
        env: dev
    spec:
      containers:
        - name: mahe
          image: nginx
          ports:
            - containerPort: 80
          env:
            - name: userid
              valueFrom:
                configMapKeyRef:
                  name: dev
                  key: userid
            - name: password
              valueFrom:
                configMapKeyRef:
                  name: dev
                  key: password
